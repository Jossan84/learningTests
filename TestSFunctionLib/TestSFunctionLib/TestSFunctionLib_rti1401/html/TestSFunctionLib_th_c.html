86/* Interface function of Task Handling to create and bind all tasks */



87static void rti_th_initialize(void)



88{



89  /* --- Local declarations ------------------------------------------------ */



90



91  /* Pointers to task control blocks */



92  rtk_p_task_control_block pTask1;  /*  Task (TCB pointer).              */



93



94  int subentry;        /*  RTK subentry.                    */



95  int service;         /*  RTK service.                     */



96



97/* Initialize dynamically generated services */



98



99  /* --- Initialization code -----------------------------------------------



100   * Task  1 : Timer Task 1 (TIMER TIMERA)



101   * Priority: 1, Source: 1, Target: 1



102   * Source IntNo: 0, SubIntNo: RTK_NO_SINT, TaskId: 0



103   * ----------------------------------------------------------------------- */



104  service   = S_PERIODIC_A;                     /*  RTK service.                     */



105  subentry = rtk_get_subentry( /* --- Get RTK subentry. ----------- */



106      service,                 /*  RTK service.                     */



107      0,                 /*  Board base address.              */



108      0);                /*  Interrupt number.                */



109  pTask1   = rtith_create_task( /* --- Create task. ---------------- */



110      rti_TIMERA,                 /*  Task function pointer.           */



111      1,                 /*  Task priority.                   */



112      ovc_fcn,                 /*  RTK overrun check type.          */



113      rti_default_overrun_fcn,                 /*  Overrun handler function.        */



114      1,                 /*  Overrun count limit.             */



115      0);                /*  Simulink TID.                    */



116  rtk_task_name_set( /* --- Set task name. -------------- */



117      pTask1,          /*  Task (TCB pointer).              */



118      "Timer Task 1");       /*  Task name.                       */



119  rtith_bind_interrupt( /* --- Bind interrupt to task. ----- */



120      service, subentry,         /*  RTK service, RTK subentry.       */



121      pTask1,             /*  Task (TCB pointer).              */



122      (1 * RTI_TIMER_TASK_TIME_SCALE),             /*  Sample time or period.           */



123      C_LOCAL,             /*  RTK channel.                     */



124      -1,             /*  Logical interrupt number.        */



125      NULL);            /*  Hook function.                   */



126  rtith_set_task_type( /* --- Set RTK task type. ---------- */



127      service, subentry,           /*  RTK service, RTK subentry.       */



128      RTK_NO_SINT,               /*  Sub-interrupt number.            */



129      rtk_tt_periodic,               /*  RTK task type.                   */



130      NULL,               /*  Reference task (time stamping).  */



131      0,            /*  Sample time offset.              */



132      1);           /*  Step multiple.                   */



133



134  /* ... Assign task information variables ................................. */



135  pRti_TIMERA_STime       = &(pTask1->sample_time);



136  pRti_TIMERA_TTime       = &(pTask1->turnaround_time);



137  pRti_TIMERA_TState      = &(pTask1->state);



138  pRti_TIMERA_OType       = &(pTask1->ovc_type);



139  pRti_TIMERA_OMax        = &(pTask1->ovc_max);



140  pRti_TIMERA_ORpt        = &(pTask1->ovc_repeat);



141  pRti_TIMERA_OCnt        = &(pTask1->ovc_counter);



142  pRti_TIMERA_TCnt        = &(pTask1->tm_task_calls);



143  pRti_TIMERA_Prio        = &(pTask1->priority);



144



145  /* ... Assign pointer to RTK task control block of of 'Timer Task 1' ..... */



146  pRtiTimerTask1TCB = pTask1;         /*  Reference task (time stamping).  */



147



148  /* ... Mark driving interrupt as unused in non-RT simulation mode ........ */



149# ifdef SMODE



150#   if SMODE == NRTSIM



151  rtith_int_status_bit_clear( /* --------------------------------- */



152    service, subentry,                 /*  RTK service, RTK subentry.       */



153    RTK_NO_SINT,                     /*  Sub-interrupt number.            */



154    RTK_STATUS_USED);       /*  RTK mask: Interrupt is bound.    */



155#   endif



156# endif



157



158  return;



159}

161void rti_th_timertask1_trigger_source(RtiTimerTask1TriggerSource* triggerSource)



162{



163  triggerSource->service = S_PERIODIC_A;



164  triggerSource->subentry = rtk_get_subentry(



165    S_PERIODIC_A,



166    0,



167    0);



168  triggerSource->subsubentry = RTK_NO_SINT;



169}
